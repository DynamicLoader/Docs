266 // Read image data

268 {
269 RNAMImage::Channel chan;
270 in.read((char *)(&chan.stat), sizeof(RNAMImage::Stat));
271

286 c.setGrey(tmp);

290
291 size_t tmp;
292 in.read((char *)(&tmp), sizeof(tmp));
293
294 for (int j = 0; j < tmp; j++)

296 char tmp2;

298 chan.coordinateList.push_back(tmp2);

300 img.data.push_back(chan);

302
303 return 0;
