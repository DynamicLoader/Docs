206 Q_INVOKABLE void startEncode(bool loopMode = false, const cv::Mat &sourceMat = cv::Mat())
207 {
208 if (_encRunner)
209 {
210 emit encodeFinished(ERRORNO_EX(TNAM,ERR_BUSY)); // Already running
211 return;
212 }
213
214 cv::Mat mt;
215 if (loopMode)
216 {
217 mt = this->_dec.restructed_img;
218 }
219 else
220 {
221 if (!sourceMat.empty())
222 {
223 mt = sourceMat;
224 }
225 else
226 {
227 mt = cv::imread(this->_inFp.toLocal8Bit().data(), cv::IMREAD_GRAYSCALE);
228 if (!mt.data)
229 {
230 emit encodeFinished(ERRORNO_EX(TNAM,ERR_EMPTY_INPUT)); // Cannot read image: no such file or unsupported format
231 return;
232 }
233 }
234 }
235
236 if(mt.empty()){
237 emit encodeFinished(ERRORNO_EX(TNAM,ERR_EMPTY_INPUT));
238 return;
239 }
240
241 // Get the binary format image to get competitable with TNAM
242 cv::Mat dst;
243 cv::threshold(mt, dst, this->_binRate * 255, 1, cv::THRESH_BINARY);
244
245
246 _encRunner = new WrapperTNAM_impl::EncoderThreadRunner(this, dst);
247 connect(_encRunner, WrapperTNAM_impl::EncoderThreadRunner::finished, this, WrapperTNAM::_onEncodeFinished);
248 _encRunner->start();

257 Q_INVOKABLE void startDecode(bool loopMode = false)
258 {
259 if (_decRunner)
260 {
261 emit decodeFinished(ERRORNO_EX(TNAM,ERR_BUSY));
262 return;
263 }
264
265 WrapperTNAM_impl::encoded_t data;
266
267 if (loopMode)
268 {
269 data = this->_enc;
270 }
271 else
272 {
273 ZIF::Coder coder(
274 [&data](std::ifstream &is, uint32_t len) -> int {
275 meta_t meta;
276 is.read((char *)&meta, sizeof(meta_t));
277 if (!is.good())
278 return -1;
279 data.line_num = meta.LineNum;
280 data.num_of_total_triangular = meta.TriNum;
281 data.point_num = meta.PointNum;
282 data.n = meta.N;
283 data.actualH = meta.actualH;
284 data.actualW = meta.actualW;
285 for (uint16_t i = 0; i < meta.TriNum; ++i)
286 {
287 dataitem_t tmp = {0};
288 is.read((char *)&tmp, sizeof(dataitem_t));
289 if (!is.good())
290 return -1;
291 data.Q_tri[0 + i] = tmp.flag;
292 data.Q_tri[65535 * 1 + i] = tmp.W;
293 data.Q_tri[65535 * 2 + i] = (tmp.X_H6 << 5) | tmp.X_L5;
294 data.Q_tri[65535 * 3 + i] = (tmp.Y_H1 << 10) | tmp.Y_L10;
295 data.Q_tri[65535 * 4 + i] = tmp.Z;
296 }
297 for (uint16_t i = 0; i < meta.LineNum + meta.PointNum; ++i)
298 {
299 dataitem_t tmp = {0};
300 is.read((char *)&tmp, sizeof(dataitem_t));
301 if (!is.good())
302 return -1;
303 data.Q[65535 * 0 + i] = tmp.W;
304 data.Q[65535 * 1 + i] = (tmp.X_H6 << 5) | tmp.X_L5;
305 data.Q[65535 * 2 + i] = (tmp.Y_H1 << 10) | tmp.Y_L10;
306 data.Q[65535 * 3 + i] = tmp.Z;
307 }
308 return 0;
309 },
310 ZIF_CODER_NO_OUTPUT, TNAM_ENCODE_ZIF);
311
312 auto ret = coder.readFile(this->_inFp.toLocal8Bit().data());
313 if (ret != 0)
314 {
315 emit decodeFinished(ret);
316 return;
317 }
318 }
319
320 _decRunner = new WrapperTNAM_impl::DecoderThreadRunner(this, data);
321 connect(_decRunner, WrapperTNAM_impl::DecoderThreadRunner::finished, this, WrapperTNAM::_onDecodeFinished);
322 _decRunner->start();
323 }

585void EncoderThreadRunner::run()
586{
587 using namespace cv;
588
589 if (!_parent)
590 {
591 emit finished(ERRORNO_EX(TNAM, ERR_NO_PARENT));
592 return;
593 }
594
595 if (_data.rows == 0 || _data.cols == 0)
596 {
597 emit finished(ERRORNO_EX(TNAM, ERR_EMPTY_INPUT)); // Empty file
598 return;
599 }
600
601 encoded_t &enc = _parent->_enc;
602 enc.n = std::max(_data.rows, _data.cols);
603 enc.actualH = _data.rows;
604 enc.actualW = _data.cols;
605
606 coder::array<boolean_T, 2U> in;
607 in.set_size(enc.n, enc.n);
608
609 for (int idx0 = 0; idx0 < enc.n; idx0++)
610 {
611 for (int idx1 = 0; idx1 < enc.n; idx1++)
612 {
613 in[idx1 * in.size(0) + idx0] =
614 (idx0 >= _data.rows || idx1 >= _data.cols ? 1 : _data.at<bool>(idx0, idx1)); // Column first
615 }
616 }
617
618 // Process
619 std::cout << enc.n << std::endl;
620 _parent->_algo.triangular_raster(enc.n, in, &enc.point_num, &enc.line_num, &enc.num_of_total_triangular, enc.Q_tri,
621 enc.Q);
622
623 emit finished(0); // No Error
624}

626void DecoderThreadRunner::run()
627{
628 if (!this->_parent)
629 {
630 emit finished(ERRORNO_EX(TNAM, ERR_NO_PARENT));
631 return;
632 }
633 if (this->_data.n == 0)
634 {
635 emit finished(ERRORNO_EX(TNAM, ERR_EMPTY_INPUT));
636 return;
637 }
638
639 // Prepare Output
640 coder::array<boolean_T, 2> res;
641 res.set_size(_data.n, _data.n);
642 // Process
643 _parent->_algo.triangular_decode(_data.n, _data.Q_tri, _data.Q, _data.num_of_total_triangular, _data.point_num,
644 _data.line_num, res);
645 decoded_t &dec = _parent->_dec;
646 (coder_base_t &)dec = (const coder_base_t &)_data;
647
648 cv::Mat resimg(dec.actualH, dec.actualW, CV_8UC1);
649 for (int i = 0; i < res.size(0) && i < dec.actualH; ++i)
650 {
651 for (int j = 0; j < res.size(1) && j < dec.actualW; ++j)
652 {
653 resimg.at<boolean_T>(i, j) = res[j * res.size(0) + i];
654 }
655 }
656
657 cv::threshold(resimg, _parent->_dec.restructed_img, 0.5, 255, cv::THRESH_BINARY);
658 emit finished(0);
659}
