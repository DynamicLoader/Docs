51static void _diff_1chan(const Mat &a, const Mat &b, Mat &output)
52{
53 if (a.channels() > 1 or b.channels() > 1)
54 return;
55
56 if (a.cols != b.cols or a.rows != b.rows)
57 return;
58
59 output = Mat::zeros(a.size(), CV_8UC1);
60 for (int i = 0; i < a.rows; i++)
61 for (int j = 0; j < a.cols; j++)
62 {
63 int diff = abs(a.at<uchar>(i, j) - b.at<uchar>(i, j));
64 output.at<uchar>(i, j) = _sigmoid(diff, diffTheshold) * 255;
65 }
66}

211 static void ImgDiffVisualizer(const cv::Mat &a, const cv::Mat &b, cv::Mat &output)
212 {
213 if (a.channels() != b.channels())
214 return;
215
216 if (a.cols != b.cols or a.rows != b.rows)
217 return;
218
219 if (a.channels() == 1)
220 {
221 _diff_1chan(a, b, output);
222 return;
223 }
224
225 output = Mat::zeros(a.size(), CV_8UC3);
226 for (int i = 0; i < a.rows; i++)
227 for (int j = 0; j < a.cols; j++)
228 for (int k = 0; k < a.channels(); k++)
229 {
230 int diff = abs(a.at<Vec3b>(i, j)[k] - b.at<Vec3b>(i, j)[k]);
231 output.at<Vec3b>(i, j)[k] = _sigmoid(diff, diffTheshold) * 255;
232 }
233 }

241 static int ScanImgType(const Mat &img)
242 {
243 bool is_gray = true;
244 bool is_binary = true;
245
246 for (int i = 0; i < img.rows; i++)
247 for (int j = 0; j < img.cols; j++)
248 {
249 if (img.at<Vec3b>(i, j)[0] != img.at<Vec3b>(i, j)[1]
250 or img.at<Vec3b>(i, j)[1] != img.at<Vec3b>(i, j)[2]
251 or img.at<Vec3b>(i, j)[0] != img.at<Vec3b>(i, j)[2])
252 return Color;
253
254 if (is_gray)
255 for (int c = 0; c < 3; c++)
256 if (img.at<Vec3b>(i, j)[c] != 0 and img.at<Vec3b>(i, j)[c] != 255)
257 is_binary = false;
258
259 if (!is_gray and !is_binary)
260 return Color;
261 }
262
263 return (is_binary + is_gray);
264 }

293 Q_INVOKABLE static int imageSuffixChecker(const QString &path)
294 {
295 using std::string;
296 string s = path.toLocal8Bit().toStdString();
297 string suffix;
298
299 bool foundDot = false;
300
301 long long p = s.length() - 1;
302 while (p > -1)
303 {
304 if (s[p] == '.')
305 {
306 foundDot = true;
307 suffix = s.substr(p + 1, s.length() - p - 1);
308 break;
309 }
310 p--;
311 }
312
313 if (!foundDot)
314 return 0;
315
316 suffix = toLowerCase(suffix);
317 if (suffix == "zif")
318 return -1;
319
320 const string validSuffix[] = {"jpg", "jpeg", "png", "bmp", "tiff", "tif"};
321 for (const auto &i : validSuffix)
322 if (suffix == i)
323 return 1;
324 return 0;
325 }
